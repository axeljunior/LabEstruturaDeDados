#include <stdio.h>
#include <stdlib.h>
/*
3) Fazer um programa com as seguintes características:
a) Contém uma função que recebe dois números inteiros;
b) A função retorna 1 se os números são iguais, 0 se são diferentes. Além disso, a
função calcula a soma e o produto entre os dois números.
c) Na main(), leia os dois números, chame a função e mostre se os números são
iguais ou não. Além disso, mostre sua soma e seu produto.
*/


int iguais(int x,int y,int *soma,int *produto)
{
	*soma=x+y;
	*produto=x*y;
	
	if(x==y)
		return(1);
	else
		return(0);
}
int main()
{
	int a,b,soma,produto;
	
	printf("\n2 numeros:");
	scanf("%d,%d",&a,&b);
	
	if(iguais(a,b,&soma,&produto))
		printf("\nSao iguais.\n");
	else
		printf("\nSão diferentes.\n");
		
	printf("Soma: %d",soma);
	printf("Produto: %d",produto);
}
